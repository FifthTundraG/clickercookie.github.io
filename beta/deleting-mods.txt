mods.deleteClicked = function() {
    helper.popup.createAdvanced(400,300,"<h3 class='simple-popup-title' style='display:block;'>Delete Mod</h3> \
    <p class='popup-content no-display' id='noModsMessage' style='font-size:13px; margin-top:7px; margin-bottom:0px;'>You have no mods installed!</p> \
    <div id='modsList' class='mods-list'></div> \
    <button onclick='helper.popup.destroyAdvanced()' id='simplePopupButton' class='popup-button' style='margin-top:20px;'>OK</button>");
    const modTags = document.getElementsByTagName("script");
    const modsToDisplay = modTags.length - 1;

    let modsLeft = modsToDisplay;
    while (modsLeft >= 1) {
        const newModDisplay = document.createElement("div");
        newModDisplay.setAttribute("class","popup-content mod-in-list");
        newModDisplay.setAttribute("id","modList" + modsLeft);

        const newModName = document.createElement("p");
        const currentModId = "mod" + modsLeft;
        newModName.appendChild(document.createTextNode(JSON.stringify(document.getElementById(currentModId).src)));
        newModName.setAttribute("class","popup-content");
        newModDisplay.appendChild(newModName);

        const xPhoto = document.createElement("img");
        xPhoto.setAttribute("src","img/red-x.png");
        xPhoto.setAttribute("alt","delete button");
        xPhoto.setAttribute("class","delete-mod-button");
        xPhoto.setAttribute("onclick","mods.delete('mod" + modsLeft + "')");
        newModDisplay.appendChild(xPhoto);

        document.getElementById("modsList").appendChild(newModDisplay);
        modsLeft -= 1;
    }
    
    if (modsToDisplay == 0) {document.getElementById("noModsMessage").style.display = "block";}
}
mods.delete = function(script) {
    const allScripts =  document.getElementsByTagName("script");
    const moddedScripts = allScripts.length - 1;

    const modNumber = script.replace("mod","");

    if (moddedScripts > 1 && modNumber != moddedScripts.length) {
        helper.popup.createAdvanced(300,150,"<p class='popup-content'>You found a glitch! I don't know how to fix it! So I gave you this popup instead as a way to hide it from view.</p>");
    }

    const scriptToDelete = document.getElementById(script);
    scriptToDelete.remove();

    const modDiv = document.getElementById("modList" + modNumber);
    modDiv.remove();

    mods.numberLoaded -= 1;
    if (mods.numberLoaded <= 0) {
        document.getElementById("noModsMessage").style.display = "block";
    }
    mods.reloadModsLoadedText();
}
